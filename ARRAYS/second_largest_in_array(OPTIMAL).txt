================================
SECOND LARGEST ELEMENT IN ARRAY
===============================

TEST CASE-1:

INPUT: 5
       1 8 3 6 7

OUTPUT: 7
==============

TEST CASE-2:

INPUT: 3
       98 100 99
OUTPUT: 99
==============


Program:
=======


#include<bits/stdc++.h>
using  namespace std;

int main()
{
  int n;
  cin>>n;
  int arr[n];
  for(int i=0;i<n;i++)
  {
    cin>>arr[i];  //reading the array elements
  }

  int max_ele=INT_MIN; // iam assuming that maximum element is minimum integer
  int sec_max=-1;      // iam assuming that second maximum element as -1

  
  for(int i=0;i<n;i++) 
  {
     if(arr[i]>max_ele)
     {
	     sec_max=max_ele;
       max_ele=arr[i];
     }
     else if(arr[i]>sec_max && arr[i]!=max_ele)
     {
       sec_max=arr[i];
     }
  }
  cout<<sec_max;//printing second maximum element
  return 0;
} 

===========================
TIME AND SPACE COMPLEXITY

TC: O(n)
SC: O(1)
=========================
