======================
LINEAR SEARCH
=====================

Example -1:

Input: arr[] = [1,5,9,3,2], n=5
	target= 5
Output: true

Example  -2:

Input:  arr[] = [1,5,9,3,2], n=5
	target= 11
Output: false


Program:
========

#include<bits/stdc++.h>
using namespace std;

bool linear_search(int arr[],int n,int target)
{
  for(int i=0;i<n;i++)
  {
    if(arr[i]==target)
	return true;
  }
 return false;
}
int main()
{
  int n;
  cin>>n;
  int arr[n];
  for(int i=0;i<n;i++)
  {
     cin>>arr[i];
  }
  int target;
  cin>>target;
  cout<<linear_search(arr,n,target);
return 0;
}


==========
TC: O(N)
SC:O(1) // iam not using any extra space to solve problem
=========
